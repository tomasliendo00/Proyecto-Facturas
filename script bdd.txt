CREATE VIEW [dbo].[VistaFacturas]
as
SELECT f.nro_factura, f.fecha, fp.id 'ID Forma Pago', fp.nombre 'Forma Pago', c.id 'ID Cliente', c.nombre 'Nombre', c.apellido 'Apellido'
FROM Facturas f 
JOIN FormasPago fp ON fp.id = f.id_forma_pago
JOIN Clientes c ON c.id = f.id_cliente

CREATE PROCEDURE [dbo].[SP_GET_BY_ID]
@id int
AS
BEGIN
SELECT f.nro_factura, f.fecha, fp.id AS [ID Forma Pago], fp.nombre AS [Forma Pago], c.id AS [ID Cliente], c.nombre, c.apellido
FROM dbo.Facturas AS f 
INNER JOIN dbo.FormasPago AS fp ON fp.id = f.id_forma_pago 
INNER JOIN dbo.Clientes AS c ON c.id = f.id_cliente
WHERE f.nro_factura = @id
END;
GO
